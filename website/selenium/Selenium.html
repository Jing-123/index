2、窗口切换

xxxxxxxxxx
from selenium import webdriver
​
browser = webdriver.Chrome(executable_path ="D:\\Python\\selenium_needs\\chromedriver") #指定驱动运行地址
browser.get("https://www.baidu.com/") #通过浏览器向服务器发送URL请求
js='window.open("http://www.testclass.net/");' #通过js打开新的窗口
browser.execute_script(js) #运行JavaScript语句
​
print('当前句柄为：') #输出当前窗口的句柄
print(browser.current_window_handle)
​
handles = browser.window_handles #获取当前窗口句柄集合【列表类型】
print('当前句柄集合为：')
print(handles)
​
#切换到当前窗口
for handle in handles:
   if handle != browser.current_window_handle:
      print('\nswitch to ',handle)
      browser.switch_to_window(handle)
      print(browser.current_window_handle)
      break
​
browser.close()
browser.switch_to_window(handles[0])

3、执行鼠标操作

x
from selenium import webdriver
​
# 引入 ActionChains 类
from selenium.webdriver.common.action_chains import ActionChains
​
browser = webdriver.Chrome(executable_path ="D:\\Python\\selenium_needs\\chromedriver") #指定驱动运行地址
browser.get("http://www.testclass.net/") #通过浏览器向服务器发送URL请求
​
# 定位需要执行操作的元素
element = browser.find_element_by_xpath('//*[@id="post-index"]/header/div/div/a[2]/div[1]/img')
# 对定位到的元素执行鼠标悬停操作
ActionChains(browser).move_to_element(element).perform()
browser.refresh()
​
element2 = browser.find_element_by_xpath('//*[@id="post-index"]/header/div/div/a[4]/div[1]/img')
# 对元素执行右击操作
ActionChains(browser).context_click(element2).perform()
browser.refresh()
​
element3 = browser.find_element_by_xpath('//*[@id="post-index"]/header/div/div/a[4]/div[1]/img')
# 对元素执行双击操作
try:
    ActionChains(browser).double_click(element3).perform()
    print("操作成功！")
except Exception:
    print("运行失败！")
​
browser.close()
